冒泡排序
a = {
	'0': 4,
	'1': 6,
	'2': 3,
	'3': 2,
	'4': 1,
	'length': 5
}


轮数 第几个最大值
1		3
2		2
3		1
4		0

loop = 1
while(loop < a['length']) {
	index = 0;
	while(index <= a['length'] - 1 - loop) {
		(a[index] > a[index + 1]) && swap()
		index <- index + 1
	}
	loop <- loop + 1
}
